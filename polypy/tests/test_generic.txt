def test_pbc():
    a, b = ge.pbc(10, 9, 20)
    c, d = ge.pbc(1, 40, 20)
    expected_a = False
    expected_b = 10
    expected_c = True
    expected_d = 41
    assert a == expected_a
    assert b == expected_b
    assert c == expected_c
    assert d == expected_d
